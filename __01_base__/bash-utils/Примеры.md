```shell
# Поиск логов за 7 дней с сообщением "error"
find /var/log -type f -mtime -7 -print0 | xargs -0 grep -li "error"
# Как найти все файлы, изменённые за последние 7 дней, и удалить их, кроме тех, что находятся в /etc/?
## (путь | исключение директории | или |) | тип    | меньше 7 дней | команда
find /     -path /etc -prune       -o      -type f   -mtime -7       -exec rm -i {} \;
# На сервере закончилось место на /. Как найти 5 самых больших файлов и очистить их?
find / -xdev -type f -size +100M -printf "%s\t%p\n" | sort -r | head -5
# Как вывести 10 самых частых HTTP-статусов из access.log в реальном времени?
tail -f /var/log/nginx/access.log | awk '{print $9}' | sort | uniq -c | sort -nr | head -10
# Все процессы с CPU > 50%
ps aux | awk '$3 > 50 && !/[k]worker|systemd|init/{print $2}' | xargs -r kill -9
```
