
## Просмотр
```shell
# Инфо о пользователе
cat /etc/passwd | grep ivan
id ivan

groups ivan

```

```shell
useradd ivan \
  # def домашняя директория | указать               | шаблон       | системный
  -m                        | -d /custom/home/ivan  | -k /etc/skel | -r \
  # Оболочка   | без оболочки
  -s /bin/bash | -s /dev/null \
  # основная группа | без группы | доп группы
  -g developers     | -N         | -G sudo,docker  \
  # задать uid
  -u 1500 \
  # срок действия
  -e 2025-12-31 
  # пароль
  -p '$6$salt$hash'
  
usermod ivan \
  # Изменить логин
  -l новое_имя \
  # Изменить директорию | переместить содержимое
  -d /new/home/ivan -m \
  # Изменить оболочку
  -s /bin/zsh \
  # текущая | Заменить дополнительные | добавить
  -g | -G | -aG sudo \
  # задать uid
  -u \
  # срок действия
  -e 2025-12-31  \
  
  # Pазблокировать\заблокировать
  -L | -U
  
userdel ivan \
  # Удалить домашнюю директорию
  -r
  # Принудительно если логин или процессы
  -f
```
